% Glossary entries

\newglossaryentry{gls:set:integers}{
	name={\ensuremath{\mathbb{Z}}},
	description={Set of all integers}
}

\newglossaryentry{gls:quanta}{
	name={quanta},
	description={The discrete indivisible units of energy that quantum mechanics implies for waves}
}

\newglossaryentry{gls:fermion}{
	name={fermion},
	description={Particles that don't like to live together in the same state}
}

\newglossaryentry{gls:bose:conensate}{
	name={Bose Condensate},
	description={Many Bosons in the lowest energy state}
}

\newglossaryentry{gls:dirac:sea}{
	name={Dirac Sea},
	description={All negative energy states are filled, If you want to get the lowest energy state, fill Dirac Sea with negative energy electrons.}
}


\newglossaryentry{gls:fermi:sphere}{
	name={Fermi Sphere},
	description={Electrons fill a sphere in momentum space, whose radius depends on number of electrons}
}

\newglossaryentry{gls:fermi:sea}{
	name={Fermi Sea},
	description={The electrons in the \gls{gls:fermi:sphere} are said to be in the Fermi Sea}
}

\newglossaryentry{gls:fermi:surface}{
	name={Fermi Surface},
	description={The surface of the \gls{gls:fermi:sphere} is called the Fermi Surface}
}


\newglossaryentry{gls:boson}{
	name={boson},
	description={Particles that do like to live together and condense into the same state}
}

\newglossaryentry{gls:family}{
	name={family},
	plural={families},
	description={Quarks are organized into three familes: (up, down), (charm,strange), (top,bottom). The $W^+$ can swap  $u$ with $d$, $s$ with $c$, $t$ and $b$, but it cannot swap between families}
}

\newglossaryentry{gls:ortho:hydrogen}{
	name={ortho hydrogen},
	description={Ortho hydrogen molecules are those in which the spins of both the nuclei are in the same direction\cite{libre:2023:chamistry}}
}

\newglossaryentry{gls:para:hydrogen}{
	name={para hydrogen},
	description={Molecules of hydrogen in which the spins of both the nuclei are in the opposite direction are called para hydrogen\cite{libre:2023:chamistry}}
}

\newglossaryentry{gls:phonon}{
	name = {phonon},
	description={A collective excitation in condensed matter, which is a quantization of the vibration mode in elastic structures}
}

\newglossaryentry{gls:plasmon}{
	name = {plasmon},
	description={The quantized intrinsic electronic collective oscillations that are self-sustained by the restoring force via Coulomb interaction}
}
